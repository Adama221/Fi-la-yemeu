# --- POUR HOSTINGER (SI VOUS N'AVEZ PAS L'APPLICATION NODE.JS DANS hPANEL) ---
# Si le site affiche "ERREUR SERVEUR (404) : <!DOCTYPE html>...", cela signifie
# qu'Apache (et probablement un ancien WordPress) intercepte les requêtes /api
# au lieu de votre backend Node.js.

# PLACEZ CE CODE DANS LE FICHIER .htaccess A LA RACINE DE VOTRE public_html

<IfModule mod_rewrite.c>
  RewriteEngine On
  
  # --- 1. REDIRECTION VERS LE BACKEND NODE.JS ---
  # Transférer toutes les requêtes /api vers le serveur Node (port 3000)
  RewriteCond %{REQUEST_URI} ^/api/ [NC]
  RewriteRule ^(.*)$ http://127.0.0.1:3000/$1 [P,L]

  # (Optionnel) Si vous avez des uploads :
  RewriteCond %{REQUEST_URI} ^/uploads/ [NC]
  RewriteRule ^(.*)$ http://127.0.0.1:3000/$1 [P,L]

  # --- 2. CONFIGURATION DU FRONTEND REACT ---
  RewriteBase /
  RewriteRule ^index\.html$ - [L]
  RewriteCond %{REQUEST_FILENAME} !-f
  RewriteCond %{REQUEST_FILENAME} !-d
  RewriteCond %{REQUEST_URI} !^/api/ [NC]
  RewriteRule . /index.html [L]
</IfModule>
